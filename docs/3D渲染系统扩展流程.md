## 1. 系统架构总览

```
┌──────────────────────────────────────────────────────────────┐
│  功能模块层 (features/impl/*.kt)                              │
│  @EventHandler fun onRender(event: RenderEvent.Extract) {    │
│      event.drawWireFrameBox(aabb, color)                     │
│      event.drawFilledBox(aabb, color)                        │
│      event.drawText(text, pos, scale, depth)                 │
│  }                                                           │
├──────────────────────────────────────────────────────────────┤
│  收集层 (RenderConsumer)                                      │
│  lines, filledBoxes, wireBoxes, texts 四个列表               │
├──────────────────────────────────────────────────────────────┤
│  事件调度 (EventDispatcher)                                   │
│  END_EXTRACTION → RenderEvent.Extract (收集)                 │
│  END_MAIN → RenderBatchManager.renderBatch (提交)            │
├──────────────────────────────────────────────────────────────┤
│  批处理层 (RenderBatchManager)                                │
│  extractLines / extractWireBoxes / extractFilledBoxes        │
│  → BufferBuilder → drawAllPipelines → encoder.drawIndexed    │
└──────────────────────────────────────────────────────────────┘
```

### 与 PIP 渲染系统的区别

| 维度 | PIP (2D GUI) | 3D 世界渲染 |
|------|-------------|-------------|
| 渲染目标 | GUI offscreen texture → blit 合成 | 主渲染目标 (mainRenderTarget) |
| 坐标系 | 屏幕像素坐标 | 世界坐标 (三维) |
| 事件 | `extractRenderState()` (GUI 帧) | `LevelExtractionEvents` (世界帧) |
| 管线基类 | `GUI_SNIPPET` | `LINES_SNIPPET` / `DEBUG_FILLED_SNIPPET` |
| 着色器 | 自定义 SDF 着色器 | 使用 Minecraft 内置 snippet |
| 矩阵 | `ProjMat * ModelViewMat` (正交) | 需手动计算相机空间位置矩阵 |
| 缓冲管理 | 每帧创建 + 帧末清理 | ByteBufferBuilder 复用 + 环形缓冲 |

## 2. 生命周期与事件流

```
游戏每帧渲染循环
  │
  ▼
LevelExtractionEvents.END_EXTRACTION
  │  └─ EventDispatcher → posts RenderEvent.Extract(handler, consumer)
  │      └─ 所有 @EventHandler fun onRender(event: RenderEvent.Extract) 被调用
  │          ├─ Nametags.onRender() → event.drawText(...), event.drawStyledBox(...)
  │          └─ (其他模块)
  ▼
(其他玩家、实体、方块渲染...)
  │
  ▼
LevelRenderEvents.END_MAIN  (第1个 listener)
  │  └─ EventDispatcher → posts RenderEvent.Last(handler)
  ▼
LevelRenderEvents.END_MAIN  (第2个 listener)
  │  └─ RenderBatchManager.renderBatch(context)
  │      ├─ renderConsumer 判空提前返回
  │      ├─ extractLines()     → 写入 LINES_ESP / LINES_TRANSLUCENT_ESP 的 BufferBuilder
  │      ├─ extractWireBoxes()  → 写入 LINES_ESP 的 BufferBuilder
  │      ├─ extractFilledBoxes() → 写入 QUADS_ESP 的 BufferBuilder
  │      ├─ drawAllPipelines()  → encoder.createRenderPass → 逐个 pipeline drawIndexed
  │      ├─ drawTexts()          → 单独提交 TEXT_SEE_THROUGH / TEXT_POLYGON_OFFSET
  │      └─ renderConsumer.clear()
  ▼
绘制完成 → 下一帧
```

### 事件注册源码

```kotlin
// EventDispatcher.kt
LevelExtractionEvents.END_EXTRACTION.register { handler ->
    Hkim.EVENT_BUS.post(RenderEvent.Extract(handler, RenderBatchManager.renderConsumer))
}

LevelRenderEvents.END_MAIN.register { handler ->
    Hkim.EVENT_BUS.post(RenderEvent.Last(handler))
}

LevelRenderEvents.END_MAIN.register(RenderBatchManager::renderBatch)
```

### 4.3 renderBatch 执行流程

```kotlin
fun renderBatch(context: LevelRenderContext) {
    // 1. 快速判空
    if (consumer 全空) return

    val matrix = context.poseStack()

    // 2. 提取数据到 BufferBuilder
    extractLines(consumer.lines, matrix)
    extractWireBoxes(consumer.wireBoxes, matrix)
    extractFilledBoxes(consumer.filledBoxes, matrix)

    // 3. GPU 提交
    val encoder = RenderSystem.getDevice().createCommandEncoder()
    val camState = context.levelState().cameraRenderState
    val viewRot = Quaternionf().set(camState.orientation).conjugate()
    val viewMatrix = Matrix4f()
        .rotation(viewRot)
        .translate(-camState.pos.x, -camState.pos.y, -camState.pos.z)
    drawAllPipelines(viewMatrix, encoder)
    usedPipelines.clear()

    // 4. 文字单独处理
    if (consumer.texts.isNotEmpty()) {
        drawTexts(consumer.texts, camState, viewMatrix, encoder)
    }

    // 5. 清空收集器
    consumer.clear()
}
```

### 4.4 drawPipeline 执行细节

```kotlin
private fun drawPipeline(ctx: PipelineContext, viewMatrix: Matrix4f, encoder: CommandEncoder) {
    // 1. 构建 MeshData
    val meshData = ctx.bufferBuilder.buildOrThrow()
    ctx.bufferBuilder = BufferBuilder(ctx.allocator, ctx.pipeline.primitiveTopology,
        ctx.pipeline.getVertexFormatBinding(0)!!)

    val vertexData = meshData.vertexBuffer()
    if (vertexData.remaining() == 0) { /* 清理返回 */ }

    // 2. 分配/复用 GPU vertex buffer
    val bufferSize = vertexData.remaining().toLong()
    if (ctx.vertexBuffer == null || ctx.vertexBuffer!!.size() < bufferSize) {
        ctx.vertexBuffer?.close()
        ctx.vertexBuffer = RenderSystem.getDevice().createBuffer(
            { "hkim_${ctx.pipeline.location}" },
            GpuBuffer.USAGE_VERTEX or GpuBuffer.USAGE_COPY_DST,
            bufferSize * 2          // 2倍空间避免频繁重分配
        )
    }
    encoder.writeToBuffer(ctx.vertexBuffer!!.slice(0, bufferSize), vertexData)

    // 3. 获取索引缓冲 (自动生成: QUADS/LINES 等拓扑由 RenderSystem 生成)
    val drawState = meshData.drawState()
    val seqBuffer = RenderSystem.getSequentialBuffer(ctx.pipeline.primitiveTopology)
    val indices = seqBuffer.getBuffer(drawState.indexCount())
    val indexType = seqBuffer.type()

    // 4. DynamicTransforms (viewMatrix + 白色调制 + 零偏移 + 单位纹理矩阵)
    val dynamicTransforms = RenderSystem.getDynamicUniforms()
        .writeTransform(viewMatrix, COLOR_MODULATOR, MODEL_OFFSET, TEXTURE_MATRIX)

    // 5. RenderPass + drawIndexed
    val renderPass = encoder.createRenderPass(...)
    renderPass.use { rp ->
        rp.setPipeline(ctx.pipeline)
        RenderSystem.bindDefaultUniforms(rp)
        rp.setUniform("DynamicTransforms", dynamicTransforms)
        rp.setVertexBuffer(0, ctx.vertexBuffer!!.slice())
        rp.setIndexBuffer(indices, indexType)
        rp.drawIndexed(drawState.indexCount(), 1, 0, 0, 0)
    }

    // 6. 清理
    meshData.close()
    ctx.allocator.clear()
}
```

### 4.5 顶点提取示例 — FilledBox

```kotlin
private fun extractFilledBoxes(boxes: List<BoxData>, matrices: PoseStack) {
    if (boxes.isEmpty()) return
    usedPipelines.add(CustomRenderPipelines.QUADS_ESP)
    val ctx = pipelineContexts[CustomRenderPipelines.QUADS_ESP] ?: return
    val buffer = ctx.bufferBuilder

    for (box in boxes) {
        val (x0, y0, z0) = box.aabb.minX.toFloat() etc.
        val (x1, y1, z1) = box.aabb.maxX.toFloat() etc.

        // 每个面 2 个三角形 = 6 个顶点，共 6 个面 = 36 顶点
        addTriangle(buffer, x0, y0, z1, x1, y0, z1, x1, y1, z1, box)  // Z+ 面
        addTriangle(buffer, x0, y0, z1, x1, y1, z1, x0, y1, z1, box)
        // ... 其余 5 个面
    }
}
```

> **注意**: 坐标已在 CPU 端转换为世界坐标，没有使用 positionMatrix 预乘。  
> 变换通过 DynamicTransforms 的 ModelViewMat 在 GPU 端完成。

### 6.3 提取逻辑

```kotlin
// RenderUtils.kt
private fun extractQuads(quads: List<QuadData>, matrices: PoseStack) {
    if (quads.isEmpty()) return
    usedPipelines.add(CustomRenderPipelines.QUAD_3D)
    val ctx = pipelineContexts[CustomRenderPipelines.QUAD_3D] ?: return
    val buffer = ctx.bufferBuilder

    for (quad in quads) {
        val (r, g, b, a) = quad.color
        val v = quad.vertices
        // 两个三角形构成一个四边形
        buffer.addVertex(v[0].x.toFloat(), v[0].y.toFloat(), v[0].z.toFloat()).setColor(r, g, b, a)
        buffer.addVertex(v[1].x.toFloat(), v[1].y.toFloat(), v[1].z.toFloat()).setColor(r, g, b, a)
        buffer.addVertex(v[2].x.toFloat(), v[2].y.toFloat(), v[2].z.toFloat()).setColor(r, g, b, a)

        buffer.addVertex(v[0].x.toFloat(), v[0].y.toFloat(), v[0].z.toFloat()).setColor(r, g, b, a)
        buffer.addVertex(v[2].x.toFloat(), v[2].y.toFloat(), v[2].z.toFloat()).setColor(r, g, b, a)
        buffer.addVertex(v[3].x.toFloat(), v[3].y.toFloat(), v[3].z.toFloat()).setColor(r, g, b, a)
    }
}
```

> **注意事项**:
> 1. 3D 渲染的**顶点坐标是世界坐标**（不是相机空间），变换由 `DynamicTransforms.ModelViewMat` 在 GPU 端完成
> 2. `RenderBatchManager.renderBatch()` 计算 `viewMatrix` 时，使用了 `camState.orientation` 的**共轭四元数**作为旋转
> 3. 使用 `encoder.writeToBuffer()` 而非 `buf.map()` 来上传顶点数据（PIP 系统使用 `buf.map()`）
> 4. 如果你的管线不需要经过批处理系统（比如用自定义拓扑 `TRIANGLE_STRIP`），需要添加到 `RenderBatchManager` 时做特殊处理
> 5. 文字渲染使用单独的 `drawTextsWithPipeline()` 流程，因为它需要绑定字体纹理到 `Sampler0`
> 6. 所有提取函数按照 `extractLines → extractWireBoxes → extractFilledBoxes → drawTexts` 的顺序执行，添加新类型时保持类似模式
